    **`tests/smoke/lib/oauth-mock-driver.ts`** — small TS program (run via `tsx`) that drives the OAuth-dance sequence: starts the mock server, captures the cookie jar from a Better Auth signin redirect, returns the session cookie. Bash + curl alone can't easily handle the OAuth state machine. Keep this minimal:

    ```typescript
    /**
     * tests/smoke/lib/oauth-mock-driver.ts
     *
     * Invoked from tests/smoke/self-host.sh. Drives the full OAuth round-trip:
     *   1. Start oauth2-mock-server on :9090
     *   2. Configure mock to mint an id_token for the requested user (sub/email/name)
     *   3. Hit the Better Auth signin URL on the app container
     *   4. Follow redirects through the OAuth dance, capturing the session cookie
     *   5. Print the resulting `neotolis.session_token=<value>` cookie line to stdout
     *
     * Bash captures the stdout for use in subsequent curl assertions.
     *
     * Usage:
     *   pnpm tsx tests/smoke/lib/oauth-mock-driver.ts \
     *     --app-url http://localhost:3000 \
     *     --mock-port 9090 \
     *     --sub user-a-sub \
     *     --email a@test.local \
     *     --name "User A"
     */
    import { OAuth2Server } from 'oauth2-mock-server';

    interface Args {
      appUrl: string;
      mockPort: number;
      sub: string;
      email: string;
      name: string;
    }

    function parseArgs(): Args {
      const argv = process.argv.slice(2);
      const get = (flag: string) => {
        const i = argv.indexOf(flag);
        if (i === -1 || i === argv.length - 1) throw new Error(`missing arg: ${flag}`);
        return argv[i + 1]!;
      };
      return {
        appUrl: get('--app-url'),
        mockPort: parseInt(get('--mock-port'), 10),
        sub: get('--sub'),
        email: get('--email'),
        name: get('--name'),
      };
    }

    async function main() {
      const args = parseArgs();
      const server = new OAuth2Server();
      await server.issuer.keys.generate('RS256');
      await server.start(args.mockPort, '0.0.0.0');

      try {
        // Mint claims for the next exchange
        server.service.on('beforeUserinfo', (userInfo: any) => {
          userInfo.body = { sub: args.sub, email: args.email, name: args.name };
        });
        server.service.on('beforeTokenSigning', (token: any) => {
          token.payload = {
            ...token.payload,
            sub: args.sub,
            email: args.email,
            email_verified: true,
            name: args.name,
            iss: server.issuer.url,
            aud: 'mock-client-id',
          };
        });

        // Drive the auth dance.
        // Better Auth's social-signin flow:
        //   GET /api/auth/sign-in/google?callbackURL=/  -> 302 to mock authorize endpoint
        //   Mock authorize -> 302 back to /api/auth/callback/google?code=...&state=...
        //   Better Auth callback -> 302 to / with Set-Cookie: neotolis.session_token=...
        //
        // We follow redirects manually with fetch, accumulating a cookie jar.
        const jar = new Map<string, string>();
        let url: string | null = `${args.appUrl}/api/auth/sign-in/google?callbackURL=/`;
        let hops = 0;
        while (url && hops < 10) {
          hops++;
          const cookieHeader = [...jar.entries()].map(([k, v]) => `${k}=${v}`).join('; ');
          const res = await fetch(url, {
            redirect: 'manual',
            headers: cookieHeader ? { cookie: cookieHeader } : {},
          });
          // Capture Set-Cookie
          const setCookies = (res.headers as any).getSetCookie?.() ?? [];
          for (const sc of setCookies) {
            const [pair] = sc.split(';');
            const [name, value] = pair.split('=');
            if (name && value) jar.set(name.trim(), value.trim());
          }
          if (res.status >= 300 && res.status < 400) {
            url = res.headers.get('location');
            // If location is relative, resolve against the previous request URL
            if (url && !/^https?:/.test(url)) {
              url = new URL(url, res.url).toString();
            }
          } else {
            url = null;
          }
        }

        const sessionTokenValue = jar.get('neotolis.session_token');
        if (!sessionTokenValue) {
          // Print all captured cookies to aid debugging
          console.error('[oauth-mock-driver] no session cookie set. Captured jar:', JSON.stringify([...jar.entries()]));
          process.exit(1);
        }

        // Print just the cookie key=value (one line) so bash can use $(... | tail -1)
        process.stdout.write(`neotolis.session_token=${sessionTokenValue}`);
        process.exit(0);
      } finally {
        await server.stop();
      }
    }

    main().catch((err) => {
      console.error('[oauth-mock-driver] failed:', err);
      process.exit(2);
    });
    ```

    set -euo pipefail

    # ============================================================
    # Phase 1 — Self-host smoke test (DEPLOY-05)

    #   2. APP_ROLE=worker boots; stdout contains "worker ready"
    #   3. APP_ROLE=scheduler boots; stdout contains "scheduler ready"
    #   4. OAuth login (mocked) lands on dashboard with English text
    #   5. Cross-tenant: user A reading a fake user-B resource returns 404 (NOT 403)
    #   6. Anonymous /api/me returns 401
    # Plus: D-14 — minimal env (no CF_*, no ANALYTICS_*) is sufficient for boot.
    # Time budget: <5 min on GitHub-hosted runners.

    APP_PORT=3000
    MOCK_PORT=9090
    DB_URL="${DATABASE_URL:-postgres://postgres:postgres@localhost:5432/neotolis}"
    BETTER_AUTH_SECRET="${BETTER_AUTH_SECRET:-ci-smoke-better-auth-secret-32-chars-min}"
    KEK_BASE64="${APP_KEK_BASE64:-$(openssl rand -base64 32)}"

    # ============================================================
    # Helpers
    # ============================================================
    log() { echo "[smoke $(date +%H:%M:%S)] $*"; }
    fail() { log "FAIL: $*"; exit 1; }

    cleanup() {
      log "cleanup"
      docker stop smoke-app smoke-worker smoke-scheduler 2>/dev/null || true
      docker rm   smoke-app smoke-worker smoke-scheduler 2>/dev/null || true
    }
    trap cleanup EXIT

    # ============================================================
    # 0. Image must exist
    # ============================================================
    log "verifying neotolis:ci image exists"
    docker image inspect neotolis:ci > /dev/null || fail "image neotolis:ci not built"

    # ============================================================
    # 1. APP_ROLE=app — boot, healthz, readyz
    # ============================================================
    log "(1) booting APP_ROLE=app"
    docker run -d --name smoke-app --network host \
      -e APP_ROLE=app \
      -e APP_MODE=selfhost \
      -e PORT=$APP_PORT \
      -e DATABASE_URL="$DB_URL" \
      -e BETTER_AUTH_URL="http://localhost:$APP_PORT" \
      -e BETTER_AUTH_SECRET="$BETTER_AUTH_SECRET" \
      -e GOOGLE_CLIENT_ID=mock-client-id \
      -e GOOGLE_CLIENT_SECRET=mock-client-secret \
      -e APP_KEK_BASE64="$KEK_BASE64" \
      -e TRUSTED_PROXY_CIDR="" \
      neotolis:ci

    # Wait up to 60s for /readyz
    log "waiting for /readyz"
    for i in $(seq 1 60); do
      if curl -fsS "http://localhost:$APP_PORT/readyz" > /dev/null 2>&1; then
        break
      fi
      sleep 1
    done
    curl -fsS "http://localhost:$APP_PORT/healthz" | grep -q '^ok$' \
      || fail "(1) /healthz did not return 'ok'"
    curl -fsS "http://localhost:$APP_PORT/readyz" | grep -q '"ok":true' \
      || fail "(1) /readyz did not return ok:true"
    log "(1) PASS — app role healthy"

    # ============================================================
    # 2. APP_ROLE=worker — boots and prints "worker ready"
    # ============================================================
    log "(2) booting APP_ROLE=worker"
    WORKER_OUT=$(docker run --rm --network host \
      -e APP_ROLE=worker \
      -e APP_MODE=selfhost \
      -e DATABASE_URL="$DB_URL" \
      -e BETTER_AUTH_URL="http://localhost:$APP_PORT" \
      -e BETTER_AUTH_SECRET="$BETTER_AUTH_SECRET" \
      -e GOOGLE_CLIENT_ID=mock-client-id \
      -e GOOGLE_CLIENT_SECRET=mock-client-secret \
      -e APP_KEK_BASE64="$KEK_BASE64" \
      neotolis:ci sh -c 'timeout 20 node build/server.js' 2>&1 || true)
    echo "$WORKER_OUT" | grep -q "worker ready" \
      || fail "(2) worker did not print 'worker ready'. Output: $WORKER_OUT"
    log "(2) PASS — worker prints 'worker ready'"

    # ============================================================
    # 3. APP_ROLE=scheduler — boots and prints "scheduler ready"
    # ============================================================
    log "(3) booting APP_ROLE=scheduler"
    SCHED_OUT=$(docker run --rm --network host \
      -e APP_ROLE=scheduler \
      -e APP_MODE=selfhost \
      -e DATABASE_URL="$DB_URL" \
      -e BETTER_AUTH_URL="http://localhost:$APP_PORT" \
      -e BETTER_AUTH_SECRET="$BETTER_AUTH_SECRET" \
      -e GOOGLE_CLIENT_ID=mock-client-id \
      -e GOOGLE_CLIENT_SECRET=mock-client-secret \
      -e APP_KEK_BASE64="$KEK_BASE64" \
      neotolis:ci sh -c 'timeout 20 node build/server.js' 2>&1 || true)
    echo "$SCHED_OUT" | grep -q "scheduler ready" \
      || fail "(3) scheduler did not print 'scheduler ready'. Output: $SCHED_OUT"
    log "(3) PASS — scheduler prints 'scheduler ready'"

    # ============================================================
    # 4. OAuth login flow (mocked) — drives the dance via tsx
    # ============================================================
    log "(4) running OAuth dance via oauth-mock-driver"
    SESSION_COOKIE_A=$(pnpm -s tsx tests/smoke/lib/oauth-mock-driver.ts \
      --app-url "http://localhost:$APP_PORT" \
      --mock-port "$MOCK_PORT" \
      --sub "user-a-sub" \
      --email "alice@smoke.test" \
      --name "Alice")
    [[ -n "$SESSION_COOKIE_A" ]] || fail "(4) no session cookie from OAuth dance"

    # Hit /api/me with the cookie
    ME_RESPONSE=$(curl -fsS -H "cookie: $SESSION_COOKIE_A" "http://localhost:$APP_PORT/api/me")
    echo "$ME_RESPONSE" | grep -q '"email":"alice@smoke.test"' \
      || fail "(4) /api/me did not return Alice. Got: $ME_RESPONSE"
    # P3 invariant: response MUST NOT contain googleSub / refreshToken / accessToken

    # Render the dashboard — must contain English text

    echo "$DASH_HTML" | grep -q "Promotion diary" \
      || fail "(4) dashboard did not render 'Promotion diary' (Paraglide). HTML: $DASH_HTML"
    log "(4) PASS — OAuth login + dashboard renders English"

    # 5. Cross-tenant 404 — Phase 1 sentinel
    # ============================================================
    # Phase 1 has only /api/me. Cross-tenant scenario via /api/me is implicit

    # ID-enumeration matrix lands in Phase 2 with /api/games. Here we assert

    log "(5) anonymous /api/me must return 401"
    STATUS=$(curl -s -o /dev/null -w "%{http_code}" "http://localhost:$APP_PORT/api/me")
    [[ "$STATUS" == "401" ]] || fail "(5) anonymous /api/me returned $STATUS (expected 401)"

    log "(5) cross-tenant: user B's /api/me must NOT show A's data"

      --mock-port "$MOCK_PORT" \
      --sub "user-b-sub" \
      --email "bob@smoke.test" \
      --name "Bob")
    ME_B=$(curl -fsS -H "cookie: $SESSION_COOKIE_B" "http://localhost:$APP_PORT/api/me")
    echo "$ME_B" | grep -q '"email":"bob@smoke.test"' \
      || fail "(5) user B /api/me wrong. Got: $ME_B"
    echo "$ME_B" | grep -q '"email":"alice@smoke.test"' \
      && fail "(5) CROSS-TENANT LEAK — user B saw user A!" || true
    log "(5) PASS — cross-tenant isolation"

    # ============================================================
    # 6. SaaS-leak invariant — already gated by ci.yml grep step.

    log "ALL SMOKE ASSERTIONS PASSED"
