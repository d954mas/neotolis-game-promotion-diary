    ```typescript
    import PgBoss from 'pg-boss';
    import { env } from './config/env.js';
    import { logger } from './logger.js';
    import { declareAllQueues } from './queues.js';

    /**
     * Create + start a pg-boss instance.
     *
     * pg-boss 10.x notes (RESEARCH.md drift table):
     *   - Use a separate connectionString (NOT a shared pg.Pool) — pg-boss
     *     manages its own connection pool internally.
     *   - createQueue() must be called for every queue before send/work (10.x
     *     change vs 9.x lazy creation).
     */
    export async function createBoss(): Promise<PgBoss> {
      const boss = new PgBoss({
        connectionString: env.DATABASE_URL,
        // Conservative defaults; Phase 3 tunes per workload
        max: 4,
        retentionDays: 30,
        archiveCompletedAfterSeconds: 3600,
        // pg-boss creates its own schema (default: 'pgboss'); migrations run on .start()
      });
      boss.on('error', (err) => {
        logger.error({ err }, 'pg-boss error');
      });
      await boss.start();
      // Phase-1 specific pitfall mitigation: declare all queues at boot
      await declareAllQueues(boss);
      return boss;
    }

    /**
     * Graceful drain: wait up to 60s for in-flight jobs (D-22 / Phase 1 surface
     * of POLL-06 which is a Phase 3 requirement). pg-boss 10.x API:
     *   boss.stop({ wait: true, graceful: true, timeout: 60_000 })
     */
    export async function stopBoss(boss: PgBoss): Promise<void> {
      logger.info('pg-boss draining');
      await boss.stop({ wait: true, graceful: true, timeout: 60_000 });
      logger.info('pg-boss stopped');
    }
    ```

    ```typescript
    import { pool } from '../lib/server/db/client.js';
    import { logger } from '../lib/server/logger.js';
    import { QUEUES } from '../lib/server/queues.js';

    /**
     * APP_ROLE=worker entrypoint. Replaces the Plan 06 stub at this same path.
     * D-01 locks src/worker/index.ts as the canonical worker module location.
     */
    export async function startWorker(): Promise<void> {
      const boss = await createBoss();

      // Phase 1: NO real handlers. Phase 3 lands poll.* handlers; Phase 1 only
      // attaches a no-op handler to internal.healthcheck so the worker has at
      // least one queue subscription (D-15 / Phase-1 pitfall: the worker must
      // be alive and healthy, not silently idle without subscriptions).
      await boss.work(QUEUES.INTERNAL_HEALTHCHECK, async (jobs) => {
        // Phase 1 no-op acknowledger. Phase 3 will replace with real work.
        for (const job of jobs) {
          logger.debug({ jobId: job.id, queue: QUEUES.INTERNAL_HEALTHCHECK }, 'healthcheck job processed');
        }
      });

      // D-15 smoke assertion #2 — exact string "worker ready" on stdout.
      // Use logger so JSON in prod / pretty in dev, but ALSO write a plain line
      // for the smoke test grep — Pino JSON contains the string but the smoke
      // test in Plan 10 should grep for "worker ready" specifically.
      // Note: Plan 06 shipped a stub printing "worker stub ready"; this plan
      // promotes the message to "worker ready" (the Plan-10 grep target).
      logger.info({ role: 'worker' }, 'worker ready');
      console.log('worker ready');

      // Graceful shutdown
      const shutdown = async (signal: string) => {
        logger.info({ signal }, 'worker received shutdown signal');
        try {
          await stopBoss(boss);
        } catch (err) {
          logger.warn({ err }, 'pg-boss stop failed');
        }
        try {
          await pool.end();
        } catch (err) {
          logger.warn({ err }, 'pool.end failed');
        }
        process.exit(0);
      };
      process.on('SIGTERM', () => shutdown('SIGTERM'));
      process.on('SIGINT', () => shutdown('SIGINT'));

      // Worker idles forever — pg-boss handles the polling loop. Block on a
      // never-resolving promise so the process stays alive.
      return new Promise<void>(() => {
        /* never resolves — process lives until SIGTERM */
      });
    }
    ```
