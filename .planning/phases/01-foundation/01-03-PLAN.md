    **`src/lib/server/db/migrate.ts`** — programmatic `migrate()` with advisory lock (RESEARCH.md Pattern 1). Exports `runMigrations()` and the `migrationsApplied` mutable flag for `/readyz` (Open Question Q4):

    ```typescript
    import { drizzle } from 'drizzle-orm/node-postgres';
    import { migrate } from 'drizzle-orm/node-postgres/migrator';
    import { Pool } from 'pg';
    import { env } from '../config/env.js';
    import { logger } from '../logger.js';

    /** Set to true once migrations finish on this process. /readyz reads this. */
    export const migrationsApplied = { current: false };

    // Stable 64-bit advisory lock key. 'MIGRATE1' as ASCII bytes — distinct from app-data locks.
    // 0x4D49475241544531 = 5_494_251_782_888_259_377 (within int8 max 9_223_372_036_854_775_807)
    const LOCK_KEY = 0x4D49475241544531n;   // BIGINT-safe; pg accepts numeric/bigint

    export async function runMigrations(): Promise<void> {
      const pool = new Pool({ connectionString: env.DATABASE_URL, max: 2 });
      const client = await pool.connect();
      try {
        logger.info({ phase: 'migrate' }, 'acquiring advisory lock');
        await client.query(`SELECT pg_advisory_lock($1)`, [LOCK_KEY.toString()]);
        const localDb = drizzle(client);
        await migrate(localDb, { migrationsFolder: './drizzle' });
        migrationsApplied.current = true;
        logger.info({ phase: 'migrate' }, 'migrations applied');
      } finally {
        try {
          await client.query(`SELECT pg_advisory_unlock($1)`, [LOCK_KEY.toString()]);
        } catch (err) {
          logger.warn({ err }, 'advisory unlock failed (non-fatal)');
        }
        client.release();
        await pool.end();
      }
    }
    ```

    **`src/lib/server/audit.ts`** — append-only writer. `writeAudit` MUST sanitize metadata (P19: never include another tenant's identifiers). The writer is INSERT-only; it intentionally has no `update` or `delete` exports.

    ```typescript
    import { db } from './db/client.js';
    import { auditLog } from './db/schema/audit-log.js';
    import { logger } from './logger.js';

    export interface AuditEntry {
      userId: string;
      action: string;                                     // e.g. 'session.signin', 'session.signout', 'session.signout_all'
      ipAddress: string;                                  // resolved IP from trusted-proxy middleware (D-19)
      userAgent?: string;
      metadata?: Record<string, unknown>;
    }

    /**
     * Append a row to audit_log. INSERT-only by design (P19/P19): the audit table is
     * the security ground truth. The app role MUST NOT have UPDATE/DELETE grants on
     * this table — enforce in deploy docs (Phase 6) but the writer never offers it.
     *
     * Metadata sanitization: keys named like other tenants (e.g. 'targetUserId') MUST
     * be omitted by the caller. We do not introspect — that would create a different
     * leak. The convention is: callers pass only their own tenant's data.
     */
    export async function writeAudit(entry: AuditEntry): Promise<void> {
      try {
        await db.insert(auditLog).values({
          userId: entry.userId,
          action: entry.action,
          ipAddress: entry.ipAddress,
          userAgent: entry.userAgent ?? null,
          metadata: entry.metadata ?? null,
        });
      } catch (err) {
        // Never let an audit failure break the user-facing request.
        // But log loudly — silent audit drops are a P19/P20 risk.
        logger.error({ err, action: entry.action, userId: entry.userId }, 'audit write failed');
      }
    }
    ```
