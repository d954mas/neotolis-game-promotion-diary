    afterEach(async () => {
      // Truncate every public table — order-insensitive via TRUNCATE ... CASCADE.
      // Skip pg_catalog and pg_toast.
      try {
        const { rows } = await pool.query<{ tablename: string }>(
          `select tablename from pg_tables where schemaname = 'public'`,

        if (rows.length === 0) return;
        const names = rows.map(r => `"${r.tablename}"`).join(', ');

        // pre-migration: nothing to truncate

    **`Dockerfile`** — multi-stage per D-22. Three stages: `deps` (install pnpm + production deps), `build` (compile SvelteKit + bundle worker entrypoints), `runtime` (copy build output and prod node_modules; non-root user). The CONTEXT chose `node:22-alpine` over distroless (D-22 explicit). Final ENTRYPOINT dispatches on `APP_ROLE`.

    ```dockerfile
    # syntax=docker/dockerfile:1.7

    # ---------- deps stage ----------
    FROM node:22-alpine AS deps
    WORKDIR /app
    RUN corepack enable && corepack prepare pnpm@9.15.0 --activate
    COPY package.json pnpm-lock.yaml* ./
    RUN pnpm install --frozen-lockfile

    # ---------- build stage ----------
    FROM deps AS build
    COPY . .
    RUN pnpm run build
    # Prune dev deps for runtime
    RUN pnpm prune --prod

    # ---------- runtime stage ----------
    FROM node:22-alpine AS runtime
    WORKDIR /app
    # non-root user (D-22 unprivileged UID; pick 10001 — well above the alpine reserved range)
    RUN addgroup -S app -g 10001 && adduser -S app -G app -u 10001
    ENV NODE_ENV=production
    COPY --from=build --chown=app:app /app/build ./build
    COPY --from=build --chown=app:app /app/node_modules ./node_modules
    COPY --from=build --chown=app:app /app/package.json ./package.json
    COPY --from=build --chown=app:app /app/drizzle ./drizzle
    USER app
    EXPOSE 3000
    HEALTHCHECK --interval=30s --timeout=5s --start-period=20s --retries=3 \
      CMD wget --quiet --spider http://localhost:3000/readyz || exit 1
    # APP_ROLE dispatch (D-15): the same build/server.js boots the right role per env
    ENTRYPOINT ["node", "build/server.js"]
    ```
