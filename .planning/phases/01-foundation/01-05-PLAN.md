    /** Seed a user + session directly (skip OAuth dance). Returns session token cookie value. */
    export async function seedUserDirectly(opts: { email: string; name?: string }): Promise<{
      userId: string;
      sessionToken: string;
      sessionCookie: string;
    }> {
      const userId = uuidv7();
      await db.insert(user).values({
        id: userId,
        email: opts.email,
        name: opts.name ?? opts.email.split('@')[0],
        emailVerified: true,
      });
      const sessionToken = randomBytes(32).toString('base64url');
      const sessionId = uuidv7();
      const expiresAt = new Date(Date.now() + 30 * 24 * 60 * 60 * 1000);
      await db.insert(session).values({
        id: sessionId,
        userId,
        token: sessionToken,
        expiresAt,
      });
      // Better Auth cookie name format: "{cookiePrefix}.session_token"
      // For tests we set the cookie directly; Plan 06's hook reads via Better Auth.
      const sessionCookie = `neotolis.session_token=${sessionToken}; Path=/; HttpOnly; SameSite=Lax`;
      return { userId, sessionToken, sessionCookie };

    export async function fetchAs(
