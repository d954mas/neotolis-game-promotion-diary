```typescript
  it("PATCH with gameIds=[g1.id, g2.id] round-trips through getEventById", async () => {
    const { userId, sessionCookie } = await setupUserWithSession();
    const game1 = await createGame(userId, "G1");
    const game2 = await createGame(userId, "G2");
    const ev = await createEvent(userId, {
      kind: "other",
      title: "Press article covering both games",
      occurredAt: new Date().toISOString(),
    }, "127.0.0.1");
    // Initial: zero junction rows.
    let dto = await getEventById(userId, ev.id);
    expect(dto.gameIds).toEqual([]);
    // PATCH /attach with multi-element array.
    const res = await fetchAuthed(`/api/events/${ev.id}/attach`, {
      method: "PATCH",
      cookie: sessionCookie,
      body: { gameIds: [game1.id, game2.id] },
    });
    expect(res.status).toBe(200);
    // Round-trip: getEventById sees both gameIds (order-agnostic — junction
    // INNER JOIN order is implementation-defined).
    dto = await getEventById(userId, ev.id);
    expect(new Set(dto.gameIds)).toEqual(new Set([game1.id, game2.id]));
    // Reverse: PATCH with empty array detaches both.
    const res2 = await fetchAuthed(`/api/events/${ev.id}/attach`, {
      method: "PATCH",
      cookie: sessionCookie,
      body: { gameIds: [] },
    });
    expect(res2.status).toBe(200);
    dto = await getEventById(userId, ev.id);
    expect(dto.gameIds).toEqual([]);
  });
});
```
