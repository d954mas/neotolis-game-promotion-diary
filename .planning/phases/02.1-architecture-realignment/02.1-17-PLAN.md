NOTE on YouTube `publishedAt` auto-fill (UAT gap "auto-enrich title + occurredAt"): YouTube oEmbed doesn't expose `published_at`. Two options were considered: (a) add YouTube Data API v3 `videos.list` call (requires user API key — KEYS-01 lands in Phase 3); (b) HTML-scrape `<meta itemprop="datePublished">` from the watch page (fragile, breaks when YouTube changes the markup). Plan 02.1-17 picks (c): SKIP `publishedAt` auto-fill in 2.1 — the user types the date manually as today (matching current /events/new defaults), but title + thumbnail + external_id are all auto-filled. Phase 3's KEYS-01 ships the YouTube Data API key UI; auto-fill of `publishedAt` lands then alongside the polling worker that needs the same key. Document this decision in the route file header.

    ```typescript
    export interface EnrichmentResult {
      kind: EventKind;
      externalId: string | null;
      title: string;
      occurredAt: Date | null;  // null in 2.1 (Phase 3 fills via YouTube Data API key)
      thumbnailUrl: string | null;
      sourceMatch: { id: string; isOwnedByMe: boolean } | null;  // for author_is_me inheritance
    }

    export async function enrichFromUrl(
      userId: string,
      url: string,
    ): Promise<EnrichmentResult> {
      const { parseIngestUrl } = await import("./url-parser.js");
      const { fetchYoutubeOembed } = await import("../integrations/youtube-oembed.js");
      const parsed = parseIngestUrl(url);
      // unsupported / reddit_deferred / twitter_post / telegram_post all throw the
      // same AppError shapes as createEventFromPaste (preserve existing UX).
      if (parsed.kind === "unsupported") {
        throw new AppError("URL not yet supported", "unsupported_url", 422, { url });
      }
      if (parsed.kind === "reddit_deferred") {
        throw new AppError("Reddit ingest arrives in Phase 3", "reddit_pending_phase3", 422);
      }
      if (parsed.kind !== "youtube_video") {
        throw new AppError(
          `paste flow does not yet handle kind '${parsed.kind}'`,
          "kind_not_yet_functional", 422, { kind: parsed.kind },
        );
      }
      // YouTube branch: oEmbed fetch (existing 5xx → 502 / 401 → private 422 / 404 → unavailable 422).
      let oembed;
      try { oembed = await fetchYoutubeOembed(parsed.canonicalUrl); }
      catch (err) {
        throw new AppError("youtube oembed unreachable", "youtube_oembed_unreachable", 502,
          { cause: String((err as Error)?.message ?? err) });
      }
      if (oembed.kind === "private") throw new AppError("video is private", "youtube_unavailable", 422, { reason: "private" });
      if (oembed.kind === "unavailable") throw new AppError("video unavailable", "youtube_unavailable", 422, { reason: "unavailable" });
      // INGEST-03 author_is_me inheritance — preserve the existing handle-URL match.
      const matchedSource = oembed.data.authorUrl !== ""
        ? await findActiveSourceByHandleUrl(userId, oembed.data.authorUrl)
        : null;
      return {
        kind: "youtube_video",
        externalId: parsed.videoId,
        title: oembed.data.title || `YouTube video ${parsed.videoId}`,
        occurredAt: null,  // 2.1 SKIP — see PLAN objective; Phase 3 fills via YouTube Data API key
        thumbnailUrl: `https://img.youtube.com/vi/${parsed.videoId}/mqdefault.jpg`,
        sourceMatch: matchedSource ? { id: matchedSource.id, isOwnedByMe: matchedSource.isOwnedByMe } : null,
      };
    }
    ```
