```svelte
<script lang="ts">
  import type { PageData } from "./$types";
  let { data }: { data: PageData } = $props();
  // existing $state variables: editTitle, editKind, editOccurredAt, editAuthorIsMe, editUrl, editNotes, editGameId, ...
  // Plan 02.1-32 ADDS:
  let editStandalone = $state(
    (data.event.metadata as { triage?: { standalone?: boolean } } | null)?.triage?.standalone === true,
  );

  // Conflict signal: when the user has gameIds attached AND tries to toggle standalone.
  let hasAttachedGames = $derived(data.event.gameIds.length > 0);
  let standaloneConflict = $derived(editStandalone === true && hasAttachedGames);

  async function handleSubmit(e: SubmitEvent) {
    e.preventDefault();
    if (standaloneConflict) {
      return; // UI guard prevents submit; service-layer 422 (Plan 28) is defense-in-depth
    }
    // 1. Submit the main PATCH for kind/title/occurredAt/authorIsMe/url/notes/gameIds (existing fields).
    const mainRes = await fetch(`/api/events/${data.event.id}`, {
      method: 'PATCH',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify({
        kind: editKind,
        title: editTitle,
        // ... existing fields
        gameIds: editGameIds,  // post-Plan-28 multi-game shape
      }),
    });
    if (!mainRes.ok) { /* show error */ return; }

    // 2. If standalone toggle differs from loaded value, call the dedicated route.
    const initialStandalone = (data.event.metadata as { triage?: { standalone?: boolean } } | null)?.triage?.standalone === true;
    if (editStandalone !== initialStandalone) {
      const path = editStandalone ? 'mark-standalone' : 'unmark-standalone';
      await fetch(`/api/events/${data.event.id}/${path}`, { method: 'PATCH' });
    }

    // 3. Navigate back to /events/[id] read-only.
    await goto(`/events/${data.event.id}`);
  }

  let confirmDeleteOpen = $state(false);
  async function handleDeleteConfirmed() {
    const res = await fetch(`/api/events/${data.event.id}`, { method: 'DELETE' });
    if (res.ok) await goto('/feed');
  }
</script>
```
