    ```ts
    import type { PageServerLoad } from "./$types.js";
    import { error } from "@sveltejs/kit";
    import { requireUser } from "$lib/server/auth-helpers.js";
    import { getGameById } from "$lib/server/services/games.js";
    import { listSteamListingsForGame } from "$lib/server/services/game-steam-listings.js";
    import { listEventsForGame } from "$lib/server/services/events.js";
    import { listGames } from "$lib/server/services/games.js";
    import { listSources } from "$lib/server/services/data-sources.js";
    import {
      toGameDto, toGameSteamListingDto, toEventDto, toDataSourceDto,
    } from "$lib/server/dto.js";

    export const load: PageServerLoad = async (event) => {
      const user = await requireUser(event);
      const gameId = event.params.gameId;
      try {
        const game = await getGameById(user.id, gameId);
        const [listings, events, games, sources] = await Promise.all([
          listSteamListingsForGame(user.id, gameId),
          listEventsForGame(user.id, gameId),
          listGames(user.id),
          listSources(user.id),
        ]);
        return {
          game: toGameDto(game),
          listings: listings.map(toGameSteamListingDto),
          events: events.map(toEventDto),
          games: games.map(toGameDto),
          sources: sources.map(toDataSourceDto),
        };
      } catch (e) {
        // NotFoundError → 404; mirrors Phase 2 pattern
        throw error(404, "not_found");
      }
    };
    ```
