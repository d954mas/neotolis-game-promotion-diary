<script lang="ts">
  import ConfirmDialog from "./ConfirmDialog.svelte";
  import { m } from "$lib/paraglide/messages.js";

  let { listing, gameId, editMode, onChange }: {
    listing: GameSteamListingDto;
    gameId: string;
    editMode: boolean;
    onChange: () => void;
  } = $props();

  let confirmOpen = $state(false);
  let removing = $state(false);

  async function handleRemoveConfirmed() {
    if (removing) return;
    removing = true;
    try {
      const res = await fetch(`/api/games/${gameId}/listings/${listing.id}`, {
        method: "DELETE",
      });
      if (res.ok) {
        confirmOpen = false;
        onChange();  // invalidateAll
      }
    } finally {
      removing = false;
    }
  }
</script>

      type="button"
      class="remove-btn"
      aria-label={m.steam_listing_remove_aria()}
      onclick={() => (confirmOpen = true)}

  {/if}
</div>

  let duplicateError = $state<{ existingGameId: string; existingState: 'active' | 'soft_deleted' } | null>(null);

    e.preventDefault();
    duplicateError = null;

      method: 'POST',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify({ appId, label: label.trim() || undefined }),

    if (res.status === 422) {
      const body = await res.json();
      if (body.error === 'steam_listing_duplicate' && body.metadata) {
        duplicateError = {
          existingGameId: body.metadata.existingGameId,
          existingState: body.metadata.existingState,
        };
        return;
      }
    }
    if (res.ok) {
      onSuccess?.();
      // reset form
    } else {
      // existing generic error path
    }
  }
</script>
