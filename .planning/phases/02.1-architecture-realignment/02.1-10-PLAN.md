<!-- Smoke flow per CONTEXT D-11 (single happy-path; dismiss-inbox deferred to Phase 3 smoke) -->

Steps (sequential, each must succeed; fail-fast on any non-2xx):
  1. Sign in via oauth2-mock-server as userA.
  2. POST /api/games {title:"Smoke Phase 2.1"} → captures gameId.
  3. POST /api/sources {kind:"youtube_channel", handleUrl:"https://youtube.com/@smoke", isOwnedByMe:true, autoImport:true} → captures sourceId.
  4. POST /api/events with paste body (existing INGEST endpoint — Plan 02.1-05 createEventFromPaste; OR explicit POST /api/events with kind=youtube_video for smoke determinism — see Plan note below).
  5. GET /api/events → assert response contains the event from step 4 with source_id=NULL.
  6. PATCH /api/events/:eventId/attach {gameId} → 200.
  7. GET /api/games/:gameId/events → assert response contains the same eventId.
  8. Cross-tenant probes: sign in as userB; GET /api/sources/:sourceId → 404; PATCH /api/events/:eventId/attach {gameId: <userB's own game>} → 404 (event ownership wins; Pitfall 4 surface).

Plan note on step 4: oEmbed in CI requires network access to youtube.com which the smoke environment may or may not allow. Two options:
  Option A — use POST /api/events {kind:"youtube_video", title, occurredAt, url} directly (free-form path); doesn't exercise oEmbed. Faster, deterministic.
  Option B — use ingest endpoint with a real YouTube URL; relies on network. Slower; matches the user-facing flow more honestly.
  Recommendation: Option A in CI for determinism; document Option B as the manual-UAT extension.

    Step A — Build `tests/smoke/lib/phase21-flow.sh` (sourced by self-host.sh):

    ```bash
    #!/usr/bin/env bash
    # Phase 2.1 unified-flow smoke. Sourced by tests/smoke/self-host.sh after the Phase 2 sentinel.
    # Asserts the register-source → paste → see-in-feed → attach → see-in-games path end to end.
    # Requires: BASE_URL, COOKIE_JAR_A, COOKIE_JAR_B (existing helpers from self-host.sh).

    set -euo pipefail

    phase21_unified_flow() {
      local base="$1"; local cookieA="$2"; local cookieB="$3"

      echo "== Phase 2.1: register YouTube data_source =="
      local sourceResp
      sourceResp=$(curl -sf -X POST "$base/api/sources" \
        -H "Content-Type: application/json" \
        -b "$cookieA" \
        -d '{"kind":"youtube_channel","handleUrl":"https://youtube.com/@smoke-21","displayName":"Smoke source","isOwnedByMe":true,"autoImport":true}')
      local sourceId
      sourceId=$(echo "$sourceResp" | jq -r '.id')
      [[ -n "$sourceId" && "$sourceId" != "null" ]] || { echo "FAIL: source id missing"; exit 1; }
      echo "$sourceResp" | jq -e 'has("userId") | not' >/dev/null || { echo "FAIL: DataSourceDto leaked userId"; exit 1; }
      echo "  source registered: $sourceId"

      echo "== Phase 2.1: create youtube_video event (manual paste path — Option A for CI determinism) =="
      local eventResp
      eventResp=$(curl -sf -X POST "$base/api/events" \
        -H "Content-Type: application/json" \
        -b "$cookieA" \
        -d '{"kind":"youtube_video","title":"Smoke video","occurredAt":"2026-04-28T12:00:00Z","url":"https://youtube.com/watch?v=smoke21"}')
      local eventId
      eventId=$(echo "$eventResp" | jq -r '.id')
      [[ -n "$eventId" && "$eventId" != "null" ]] || { echo "FAIL: event id missing"; exit 1; }
      # source_id should be NULL on this manual-paste flow (no oEmbed match in CI).
      echo "$eventResp" | jq -e '.sourceId == null' >/dev/null || { echo "FAIL: expected sourceId=null on manual paste"; exit 1; }
      echo "  event created: $eventId"

      echo "== Phase 2.1: GET /api/events (the /feed query) returns the new event =="
      local feedResp
      feedResp=$(curl -sf "$base/api/events" -b "$cookieA")
      echo "$feedResp" | jq -e --arg id "$eventId" '.rows | map(.id) | index($id) != null' >/dev/null \
        || { echo "FAIL: /api/events did not return the new event"; exit 1; }

      echo "== Phase 2.1: create a game so we can attach =="
      local gameResp
      gameResp=$(curl -sf -X POST "$base/api/games" \
        -H "Content-Type: application/json" \
        -b "$cookieA" \
        -d '{"title":"Smoke Phase 2.1 game"}')
      local gameId
      gameId=$(echo "$gameResp" | jq -r '.id')
      [[ -n "$gameId" && "$gameId" != "null" ]] || { echo "FAIL: game id missing"; exit 1; }

      echo "== Phase 2.1: PATCH /api/events/:id/attach =="
      local attachResp
      attachResp=$(curl -sf -X PATCH "$base/api/events/$eventId/attach" \
        -H "Content-Type: application/json" \
        -b "$cookieA" \
        -d "{\"gameId\":\"$gameId\"}")
      echo "$attachResp" | jq -e --arg gid "$gameId" '.gameId == $gid' >/dev/null \
        || { echo "FAIL: attach did not set gameId on event"; exit 1; }

      echo "== Phase 2.1: GET /api/games/:gameId/events returns the attached event =="
      local gameEventsResp
      gameEventsResp=$(curl -sf "$base/api/games/$gameId/events" -b "$cookieA")
      echo "$gameEventsResp" | jq -e --arg id "$eventId" 'map(.id) | index($id) != null' >/dev/null \
        || { echo "FAIL: per-game events did not include the attached event"; exit 1; }

      echo "== Phase 2.1: cross-tenant probes — userB tries userA's resources =="
      local statusSrc
      statusSrc=$(curl -s -o /dev/null -w '%{http_code}' \
        -X GET "$base/api/sources/$sourceId" -b "$cookieB")
      [[ "$statusSrc" == "404" ]] || { echo "FAIL: cross-tenant GET /api/sources/:id status=$statusSrc (expected 404)"; exit 1; }

      # Userb creates their own game, then tries to attach userA's eventId to it — must 404 (event ownership wins).
      local userbGameResp
      userbGameResp=$(curl -sf -X POST "$base/api/games" \
        -H "Content-Type: application/json" \
        -b "$cookieB" \
        -d '{"title":"User B game"}')
      local userbGameId
      userbGameId=$(echo "$userbGameResp" | jq -r '.id')
      local statusAttach
      statusAttach=$(curl -s -o /dev/null -w '%{http_code}' \
        -X PATCH "$base/api/events/$eventId/attach" \
        -H "Content-Type: application/json" \
        -b "$cookieB" \
        -d "{\"gameId\":\"$userbGameId\"}")
      [[ "$statusAttach" == "404" ]] || { echo "FAIL: cross-tenant PATCH /api/events/:id/attach status=$statusAttach (expected 404; Pitfall 4 — must NOT be 500)"; exit 1; }

      # Negative assertion on response body — no "forbidden" or "permission" leakage (AGENTS.md invariant 2).
      local attachBody
      attachBody=$(curl -s -X PATCH "$base/api/events/$eventId/attach" \
        -H "Content-Type: application/json" \
        -b "$cookieB" \
        -d "{\"gameId\":\"$userbGameId\"}")
      echo "$attachBody" | grep -iE 'forbidden|permission' && { echo "FAIL: cross-tenant body leaked 'forbidden|permission'"; exit 1; } || true

      echo "== Phase 2.1 unified flow: PASS =="
    }
    ```
