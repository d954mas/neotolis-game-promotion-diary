Two new functions modeled after Plan 02.1-14 restoreEvent + Plan 02.1-05 dismissFromInbox:

```typescript
export async function markStandalone(
  userId: string,
  eventId: string,
  ipAddress: string,
  userAgent?: string,
): Promise<EventRow> {
  // Use jsonb_set with create_missing=true so metadata.triage parent is created
  // if absent (mirrors dismissFromInbox's nested jsonb_set pattern).
  const [row] = await db
    .update(events)
    .set({
      gameId: null, // standalone implies detached from game
      metadata: sql`jsonb_set(
        COALESCE(${events.metadata}, '{}'::jsonb),
        '{triage,standalone}',
        'true'::jsonb,
        true
      )`,
      updatedAt: new Date(),
    })
    .where(
      and(
        eq(events.userId, userId),
        eq(events.id, eventId),
        isNull(events.deletedAt),
      ),
    )
    .returning();
  if (!row) throw new NotFoundError();

  // Audit AFTER the UPDATE succeeds (Plan 02.1-14 precedent for non-destructive
  // actions; cross-tenant NotFoundError must NOT generate misleading audit).
  await writeAudit({
    userId,
    action: "event.marked_standalone",
    ipAddress,
    userAgent,
    metadata: { event_id: row.id, kind: row.kind },
  });

  return row;
}

export async function unmarkStandalone(
  userId: string,
  eventId: string,
  ipAddress: string,
  userAgent?: string,
): Promise<EventRow> {
  const [row] = await db
    .update(events)
    .set({
      metadata: sql`jsonb_set(
        COALESCE(${events.metadata}, '{}'::jsonb),
        '{triage,standalone}',
        'false'::jsonb,
        true
      )`,
      updatedAt: new Date(),
    })
    .where(
      and(
        eq(events.userId, userId),
        eq(events.id, eventId),
        isNull(events.deletedAt),
      ),
    )
    .returning();
  if (!row) throw new NotFoundError();

  await writeAudit({
    userId,
    action: "event.unmarked_standalone",
    ipAddress,
    userAgent,
    metadata: { event_id: row.id, kind: row.kind },
  });

  return row;
}
```

Privacy invariant 1 (CLAUDE.md): every Drizzle query above includes `eq(events.userId, userId)` as the FIRST clause inside `and(...)`. Cross-tenant calls return `undefined` from the UPDATE (zero rows match the userId filter), which throws NotFoundError → 404 at the HTTP boundary.
