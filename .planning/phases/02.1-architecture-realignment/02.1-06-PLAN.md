    ```ts
    import { Hono } from "hono";
    import { zValidator } from "@hono/zod-validator";
    import { z } from "zod";
    import { tenantScope } from "../middleware/tenant.js";
    import { mapErr, type RouteVars } from "./_shared.js";
    import {
      createSource, listSources, getSourceById, updateSource,
      softDeleteSource, restoreSource,
    } from "../../services/data-sources.js";
    import { toDataSourceDto } from "../../dto.js";

    const sourceKindEnum = z.enum([
      "youtube_channel", "reddit_account", "twitter_account", "telegram_channel", "discord_server",
    ]);

    const createSourceSchema = z.object({
      kind: sourceKindEnum,
      handleUrl: z.string().url(),
      displayName: z.string().min(1).max(120).optional().nullable(),
      channelId: z.string().min(1).max(64).optional().nullable(),
      isOwnedByMe: z.boolean().optional(),
      autoImport: z.boolean().optional(),
      metadata: z.record(z.unknown()).optional(),
    });

    const updateSourceSchema = z.object({
      displayName: z.string().min(1).max(120).optional().nullable(),
      autoImport: z.boolean().optional(),
      metadata: z.record(z.unknown()).optional(),
    });

    const listQuerySchema = z.object({
      includeDeleted: z.enum(["true", "false"]).optional(),
    });

    export const sourcesRouter = new Hono<{ Variables: RouteVars }>()
      .use("*", tenantScope)
      .post("/", zValidator("json", createSourceSchema), async (c) => {
        const body = c.req.valid("json");
        try {
          const row = await createSource(c.var.userId, body, c.var.clientIp, c.req.header("user-agent"));
          return c.json(toDataSourceDto(row), 201);
        } catch (e) { return mapErr(c, e); }
      })
      .get("/", zValidator("query", listQuerySchema), async (c) => {
        const includeDeleted = c.req.valid("query").includeDeleted === "true";
        try {
          const rows = await listSources(c.var.userId, { includeDeleted });
          return c.json(rows.map(toDataSourceDto), 200);
        } catch (e) { return mapErr(c, e); }
      })
      .get("/:id", async (c) => {
        try {
          const row = await getSourceById(c.var.userId, c.req.param("id"));
          return c.json(toDataSourceDto(row), 200);
        } catch (e) { return mapErr(c, e); }
      })
      .patch("/:id", zValidator("json", updateSourceSchema), async (c) => {
        try {
          const row = await updateSource(
            c.var.userId, c.req.param("id"), c.req.valid("json"),
            c.var.clientIp, c.req.header("user-agent"),
          );
          return c.json(toDataSourceDto(row), 200);
        } catch (e) { return mapErr(c, e); }
      })
      .delete("/:id", async (c) => {
        try {
          const row = await softDeleteSource(
            c.var.userId, c.req.param("id"), c.var.clientIp, c.req.header("user-agent"),
          );
          return c.json(toDataSourceDto(row), 200);
        } catch (e) { return mapErr(c, e); }
      })
      .post("/:id/restore", async (c) => {
        try {
          const row = await restoreSource(
            c.var.userId, c.req.param("id"), c.var.clientIp, c.req.header("user-agent"),
          );
          return c.json(toDataSourceDto(row), 200);
        } catch (e) { return mapErr(c, e); }
      });
    ```
