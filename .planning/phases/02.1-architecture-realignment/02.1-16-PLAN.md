  <behavior>
    - `formatFeedDate(now)` → `"Today, HH:MM"` where HH:MM is the local-time hours/minutes of `now`.
    - `formatFeedDate(yesterday-anytime)` → `"Yesterday"` (no time — coarser bucket).
    - `formatFeedDate(thisYear-pastDate)` → `"Apr 25"` (month-day, no year).
    - `formatFeedDate(priorYear-anyDate)` → `"Apr 25, 2025"` (month-day, year).
    - Edge: midnight rollover — `formatFeedDate(Date.now() - 1)` (a millisecond ago) returns "Today, …"; `formatFeedDate(Date.now() - 86400_000 - 1000)` (~24h ago) returns "Yesterday".
    - Edge: invalid input throws or returns `"—"` (deterministic graceful fallback). Choose: return `"—"` so the UI never renders "Invalid Date".

    ```typescript
    // formatFeedDate — compact / relative date format for the feed surface
    // (Plan 02.1-16 / Gap 5). The full ISO datetime was visually loud per UAT;
    // this helper buckets dates into 4 surfaces:
    //   - same calendar day → "Today, HH:MM" (24h locale)
    //   - one calendar day back → "Yesterday"
    //   - same calendar year, older than yesterday → "MMM D" (Apr 25)
    //   - earlier years → "MMM D, YYYY" (Apr 25, 2025)
    //
    // Locale: Intl.DateTimeFormat with the user's environment locale (no
    // server-side locale lock). All four buckets are stable across DST
    // transitions because we compare calendar dates, not raw millis.
    //
    // Returns "—" on invalid input so the feed never renders "Invalid Date".

    export function formatFeedDate(input: Date | string | null | undefined): string {
      if (input == null) return "—";
      const d = input instanceof Date ? input : new Date(input);
      if (Number.isNaN(d.getTime())) return "—";

      const now = new Date();
      const sameDay =
        d.getFullYear() === now.getFullYear() &&
        d.getMonth() === now.getMonth() &&
        d.getDate() === now.getDate();
      if (sameDay) {
        const hh = String(d.getHours()).padStart(2, "0");
        const mm = String(d.getMinutes()).padStart(2, "0");
        return `Today, ${hh}:${mm}`;
      }

      const yesterday = new Date(now);
      yesterday.setDate(now.getDate() - 1);
      const isYesterday =
        d.getFullYear() === yesterday.getFullYear() &&
        d.getMonth() === yesterday.getMonth() &&
        d.getDate() === yesterday.getDate();
      if (isYesterday) return "Yesterday";

      const sameYear = d.getFullYear() === now.getFullYear();
      if (sameYear) {
        return d.toLocaleDateString(undefined, { month: "short", day: "numeric" });
      }
      return d.toLocaleDateString(undefined, { month: "short", day: "numeric", year: "numeric" });
    }
    ```

    ```typescript
      it("returns 'Today, HH:MM' for same-day input", () => {
        const out = formatFeedDate(new Date("2026-04-28T09:05:00.000Z"));
        expect(out).toMatch(/^Today, \d{2}:\d{2}$/);
      });

      it("returns 'Yesterday' for one-calendar-day-back input", () => {
        const out = formatFeedDate(new Date("2026-04-27T18:00:00.000Z"));
        expect(out).toBe("Yesterday");
      });

      it("returns 'MMM D' for same-year older input", () => {
        const out = formatFeedDate(new Date("2026-04-15T12:00:00.000Z"));
        expect(out).toMatch(/^Apr 1[45]$/); // tolerant of TZ rendering
      });

      it("returns 'MMM D, YYYY' for prior-year input", () => {
        const out = formatFeedDate(new Date("2025-12-25T12:00:00.000Z"));
        expect(out).toMatch(/^Dec (24|25), 2025$/);
      });

      it("accepts ISO string input", () => {
        const out = formatFeedDate("2026-04-28T09:05:00.000Z");
        expect(out).toMatch(/^Today, \d{2}:\d{2}$/);
      });

      it("returns '—' on null / undefined / invalid input", () => {
        expect(formatFeedDate(null)).toBe("—");
        expect(formatFeedDate(undefined)).toBe("—");
        expect(formatFeedDate("not-a-date")).toBe("—");
        expect(formatFeedDate(new Date("invalid"))).toBe("—");
      });
    });
    ```
