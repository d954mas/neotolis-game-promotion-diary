    2. createEvent (extended for unified-table kinds + nullable gameId):
       ```ts
       export interface CreateEventInput {
         gameId?: string | null;
         kind: EventKind;
         occurredAt: Date;
         title: string;
         url?: string | null;
         notes?: string | null;
         metadata?: Record<string, unknown>;
         sourceId?: string | null;
         authorIsMe?: boolean;
         externalId?: string | null;
       }

       export async function createEvent(
         userId: string,
         input: CreateEventInput,
         ipAddress: string,
         userAgent: string | undefined,
       ): Promise<EventRow> {
         assertValidKind(input.kind);
         if (input.gameId != null) {
           // Pitfall 4 mitigation: verify gameId ownership BEFORE INSERT (no half-write).
           await assertGameOwnedByUser(userId, input.gameId);
         }
         try {
           const [row] = await db
             .insert(events)
             .values({
               userId,
               gameId: input.gameId ?? null,
               sourceId: input.sourceId ?? null,
               kind: input.kind,
               authorIsMe: input.authorIsMe ?? false,
               occurredAt: input.occurredAt,
               title: input.title,
               url: input.url ?? null,
               notes: input.notes ?? null,
               metadata: input.metadata ?? {},
               externalId: input.externalId ?? null,
             })
             .returning();
           if (!row) throw new Error("insert returned no row");
           await writeAudit({
             userId,
             action: "event.created",
             ipAddress,
             userAgent,
             metadata: { event_id: row.id, kind: row.kind, game_id: row.gameId },
           });
           return row;
         } catch (e: unknown) {
           if (isPgUniqueViolation(e)) {
             throw new AppError(
               "event already exists for this source",
               "duplicate_event",
               409,
               { kind: input.kind, source_id: input.sourceId, external_id: input.externalId },
             );
           }
           throw e;
         }
       }

       function assertValidKind(kind: string): asserts kind is EventKind {
         if (!(VALID_EVENT_KINDS as readonly string[]).includes(kind)) {
           throw new AppError(`unknown kind '${kind}'`, "validation_failed", 422, { field: "kind" });
         }
       }

       async function assertGameOwnedByUser(userId: string, gameId: string): Promise<void> {
         const [row] = await db
           .select({ id: games.id })
           .from(games)
           .where(and(eq(games.userId, userId), eq(games.id, gameId), isNull(games.deletedAt)));
         if (!row) throw new NotFoundError();
       }

       function isPgUniqueViolation(e: unknown): boolean {
         return typeof e === "object" && e !== null && "code" in e && (e as { code: string }).code === "23505";
       }
       ```

       export async function createEventFromPaste(
         userId: string,
         input: PasteInput,
         ipAddress: string,
         userAgent: string | undefined,
       ): Promise<EventRow> {
         const parsed = parseUrl(input.url);
         if (parsed.kind === "unsupported") {
           throw new AppError(`unsupported URL`, "unsupported_url", 422, { url: input.url });
         }
         if (parsed.kind === "reddit_post") {
           // CONTEXT DV-7: Reddit ingest stays Phase 3 with poll.reddit. Surface the message.
           throw new AppError("Reddit ingest arrives in Phase 3", "reddit_pending_phase3", 422, {});
         }
         if (parsed.kind !== "youtube_video") {
           throw new AppError(`paste flow does not yet handle kind '${parsed.kind}'`, "kind_not_yet_functional", 422, { kind: parsed.kind });
         }
         // YouTube oEmbed validation (Phase 2 precedent — discriminated union).
         const meta = await fetchYoutubeOembed(input.url);
         if (meta.kind === "private") {
           throw new AppError("video is private", "youtube_unavailable", 422, { reason: "private" });
         }
         if (meta.kind === "unavailable") {
           throw new AppError("video is unavailable", "youtube_unavailable", 422, { reason: "unavailable" });

         const matchedSource = await findSourceByAuthorUrl(userId, meta.authorUrl);
         return createEvent(
           userId,
           {

             kind: "youtube_video",
             occurredAt: meta.publishedAt ?? new Date(),
             title: meta.title,
             url: input.url,

             authorIsMe: matchedSource?.isOwnedByMe ?? false,

           ipAddress,
           userAgent,
         );

    4. listFeedPage — RESEARCH §3.3 verbatim:
       ```ts
       export async function listFeedPage(
         userId: string,
         filters: FeedFilters,
         cursor: string | null,
       ): Promise<FeedPage> {
         let parsedCursor: { at: Date; id: string } | null = null;
         if (cursor) parsedCursor = decodeCursor(cursor);

         const cursorClause = parsedCursor
           ? sql`(${events.occurredAt}, ${events.id}) < (${parsedCursor.at}, ${parsedCursor.id})`
           : sql`true`;

         const whereParts: ReturnType<typeof eq>[] = [];
         whereParts.push(eq(events.userId, userId));               // P1: userId FIRST (PITFALL P19 mitigation by construction)
         whereParts.push(isNull(events.deletedAt));

         if (filters.source !== undefined) whereParts.push(eq(events.sourceId, filters.source));
         if (filters.kind !== undefined) whereParts.push(eq(events.kind, filters.kind));
         if (filters.game !== undefined) whereParts.push(eq(events.gameId, filters.game));
         if (filters.attached === true) whereParts.push(isNotNull(events.gameId));
         if (filters.attached === false) {
           whereParts.push(isNull(events.gameId));
           // Inbox view excludes dismissed events (RESEARCH §6.2).
           whereParts.push(sql`COALESCE(${events.metadata}->'inbox'->>'dismissed', 'false') = 'false'`);
         }
         if (filters.authorIsMe !== undefined) whereParts.push(eq(events.authorIsMe, filters.authorIsMe));
         if (filters.from !== undefined) whereParts.push(gte(events.occurredAt, filters.from));
         if (filters.to !== undefined) whereParts.push(lte(events.occurredAt, filters.to));

         const rows = await db
           .select()
           .from(events)
           .where(and(...whereParts, cursorClause))
           .orderBy(sql`${events.occurredAt} DESC, ${events.id} DESC`)
           .limit(FEED_PAGE_SIZE + 1);

         const hasMore = rows.length > FEED_PAGE_SIZE;
         const page = rows.slice(0, FEED_PAGE_SIZE);
         const last = page[page.length - 1];
         return {
           rows: page,
           nextCursor: hasMore && last ? encodeCursor(last.occurredAt, last.id) : null,
         };
       }
       ```
