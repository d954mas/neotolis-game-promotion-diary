    ```ts
    import { and, eq, isNull } from "drizzle-orm";
    import { db } from "../db/client.js";
    import { dataSources } from "../db/schema/data-sources.js";
    import type { SourceKind } from "../integrations/data-source-adapter.js";
    import { AppError, NotFoundError } from "./errors.js";
    import { writeAudit } from "../audit.js";
    import { env } from "../config/env.js";

    export type DataSourceRow = typeof dataSources.$inferSelect;

    const FUNCTIONAL_KINDS: ReadonlySet<SourceKind> = new Set(["youtube_channel"]);

    const KIND_PHASE: Readonly<Record<SourceKind, string>> = {
      youtube_channel: "Phase 2.1",
      reddit_account: "Phase 3",
      twitter_account: "v2 (paid Twitter API gated)",
      telegram_channel: "Phase 5+",
      discord_server: "Phase 5+",
    };

    export interface CreateSourceInput {
      kind: SourceKind;
      handleUrl: string;
      displayName?: string | null;
      channelId?: string | null;
      isOwnedByMe?: boolean;
      autoImport?: boolean;
      metadata?: Record<string, unknown>;
    }

    export async function createSource(
      userId: string,
      input: CreateSourceInput,
      ipAddress: string,
      userAgent: string | undefined,
    ): Promise<DataSourceRow> {
      validateKind(input.kind);
      if (!FUNCTIONAL_KINDS.has(input.kind)) {
        throw new AppError(
          `kind '${input.kind}' is not yet functional (schema-only in Phase 2.1)`,
          "kind_not_yet_functional",
          422,
          { kind: input.kind, available_phase: KIND_PHASE[input.kind] },
        );
      }
      try {
        const [row] = await db
          .insert(dataSources)
          .values({
            userId,
            kind: input.kind,
            handleUrl: input.handleUrl,
            channelId: input.channelId ?? null,
            displayName: input.displayName ?? null,
            isOwnedByMe: input.isOwnedByMe ?? true,
            autoImport: input.autoImport ?? true,
            metadata: input.metadata ?? {},
          })
          .returning();
        if (!row) throw new Error("insert returned no row");
        await writeAudit({
          userId,
          action: "source.added",
          ipAddress,
          userAgent,
          metadata: { source_id: row.id, kind: row.kind, handle_url: row.handleUrl },
        });
        return row;
      } catch (e: unknown) {
        // Mirror Phase 2 items-youtube precedent: translate Postgres 23505 unique_violation
        // (data_sources_user_handle_active_unq) into a clean AppError. EXCEPTION to the
        // no-try/catch-around-INSERT rule (D-19): not a half-write cleanup, only a code translation.
        if (isPgUniqueViolation(e)) {
          throw new AppError(
            "data source already registered",
            "duplicate_source",
            422,
            { handle_url: input.handleUrl },
          );
        }
        throw e;
      }
    }

    export async function listSources(
      userId: string,
      options: { includeDeleted?: boolean } = {},
    ): Promise<DataSourceRow[]> {
      const where = options.includeDeleted
        ? eq(dataSources.userId, userId)
        : and(eq(dataSources.userId, userId), isNull(dataSources.deletedAt));
      return db.select().from(dataSources).where(where).orderBy(dataSources.createdAt);
    }

    export async function getSourceById(
      userId: string,
      sourceId: string,
    ): Promise<DataSourceRow> {
      const [row] = await db
        .select()
        .from(dataSources)
        .where(and(eq(dataSources.userId, userId), eq(dataSources.id, sourceId)));
      if (!row) throw new NotFoundError();
      return row;
    }

    export interface UpdateSourcePatch {
      displayName?: string | null;
      autoImport?: boolean;
      metadata?: Record<string, unknown>;
    }

    export async function updateSource(
      userId: string,
      sourceId: string,
      patch: UpdateSourcePatch,
      ipAddress: string,
      userAgent: string | undefined,
    ): Promise<DataSourceRow> {
      const existing = await getSourceById(userId, sourceId);
      if (existing.deletedAt !== null) throw new NotFoundError();

      const [row] = await db
        .update(dataSources)
        .set({
          ...(patch.displayName !== undefined ? { displayName: patch.displayName } : {}),
          ...(patch.autoImport !== undefined ? { autoImport: patch.autoImport } : {}),
          ...(patch.metadata !== undefined ? { metadata: patch.metadata } : {}),
          updatedAt: new Date(),
        })
        .where(
          and(
            eq(dataSources.userId, userId),
            eq(dataSources.id, sourceId),
            isNull(dataSources.deletedAt),
          ),
        )
        .returning();
      if (!row) throw new NotFoundError();

      if (patch.autoImport !== undefined && patch.autoImport !== existing.autoImport) {
        await writeAudit({
          userId,
          action: "source.toggled_auto_import",
          ipAddress,
          userAgent,
          metadata: {
            source_id: row.id,
            kind: row.kind,
            from: existing.autoImport,
            to: patch.autoImport,
          },
        });
      }
      return row;
    }

    export async function softDeleteSource(
      userId: string,
      sourceId: string,
      ipAddress: string,
      userAgent: string | undefined,
    ): Promise<DataSourceRow> {
      const existing = await getSourceById(userId, sourceId);
      if (existing.deletedAt !== null) throw new NotFoundError();

      // D-32 forensics ordering: AUDIT BEFORE the soft-delete UPDATE so the security signal
      // is captured even if the UPDATE later fails. Mirrors Phase 2 removeSteamKey precedent.
      await writeAudit({
        userId,
        action: "source.removed",
        ipAddress,
        userAgent,
        metadata: { source_id: existing.id, kind: existing.kind, handle_url: existing.handleUrl },
      });

      const [row] = await db
        .update(dataSources)
        .set({ deletedAt: new Date(), updatedAt: new Date() })
        .where(
          and(
            eq(dataSources.userId, userId),
            eq(dataSources.id, sourceId),
            isNull(dataSources.deletedAt),
          ),
        )
        .returning();
      if (!row) throw new NotFoundError();
      return row;
    }

    export async function restoreSource(
      userId: string,
      sourceId: string,
      ipAddress: string,
      userAgent: string | undefined,
    ): Promise<DataSourceRow> {
      // RETENTION_DAYS already in env (Phase 2). Refuse restore if window expired.
      const [row] = await db
        .select()
        .from(dataSources)
        .where(and(eq(dataSources.userId, userId), eq(dataSources.id, sourceId)));
      if (!row) throw new NotFoundError();
      if (row.deletedAt === null) throw new NotFoundError();
      const cutoff = new Date(Date.now() - env.RETENTION_DAYS * 86_400_000);
      if (row.deletedAt < cutoff) {
        throw new AppError(
          "retention window expired",
          "retention_expired",
          422,
          { source_id: row.id, deleted_at: row.deletedAt.toISOString() },
        );
      }
      const [restored] = await db
        .update(dataSources)
        .set({ deletedAt: null, updatedAt: new Date() })
        .where(and(eq(dataSources.userId, userId), eq(dataSources.id, sourceId)))
        .returning();
      if (!restored) throw new NotFoundError();
      // Restore audit verb is NOT in the new audit_action enum vocabulary
      // (RESEARCH §4.4 — restore is a low-risk operator action; keep audit clean).
      // If a verb is needed, file a Phase 6 housekeeping ticket; do NOT add ad-hoc here.
      return restored;
    }

    /**
     * Used by INGEST-03 author_is_me inheritance — given an oEmbed author_url,
     * return the matching active YouTube channel data_source for this user (if any).
     */
    export async function findSourceByAuthorUrl(
      userId: string,
      authorUrl: string,
    ): Promise<DataSourceRow | null> {
      const [row] = await db
        .select()
        .from(dataSources)
        .where(
          and(
            eq(dataSources.userId, userId),
            eq(dataSources.kind, "youtube_channel"),
            eq(dataSources.isOwnedByMe, true),
            eq(dataSources.handleUrl, authorUrl),
            isNull(dataSources.deletedAt),
          ),
        );
      return row ?? null;
    }

    function validateKind(kind: string): asserts kind is SourceKind {
      const valid: SourceKind[] = [
        "youtube_channel", "reddit_account", "twitter_account", "telegram_channel", "discord_server",
      ];
      if (!(valid as string[]).includes(kind)) {
        throw new AppError(
          `unknown kind '${kind}'`,
          "validation_failed",
          422,
          { field: "kind" },
        );
      }
    }

    function isPgUniqueViolation(e: unknown): boolean {
      return typeof e === "object" && e !== null && "code" in e && (e as { code: string }).code === "23505";
    }
    ```
