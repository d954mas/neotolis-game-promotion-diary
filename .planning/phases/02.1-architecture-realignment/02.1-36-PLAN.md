```typescript
// Plan 02.1-36 / UAT-NOTES.md §5.9: strip multi-line block comments
// (including JSDoc /** ... */) at the FILE level so cross-line comment
// boundaries don't survive into the per-line grep below. JSDoc is canonical
// project documentation per AGENTS.md commenting policy — references to
// `process.env` inside JSDoc must NOT trigger the env-discipline tripwire.
const stripped = src.replace(/\/\*[\s\S]*?\*\//g, "");
const lines = stripped.split(/\r?\n/);
for (let i = 0; i < lines.length; i++) {
  const line = lines[i]!;
  const lineStripped = line.replace(/\/\/.*$/, "");
  if (/\bprocess\.env\b/.test(lineStripped)) {
    offenders.push({ file: rel, line: i + 1, text: line.trim() });
  }
}
```

Note: stripping at file level shifts `i + 1` line numbers slightly when block comments span multiple lines, BUT this only affects the offender display (when a real violation IS reported, the displayed line number can be off). Since the scanner is a tripwire (offenders === 0 is the success path), the line-number drift is acceptable. If the executor wants pixel-perfect line numbers, replace each block-comment match with the same number of newlines as it consumed (e.g., `match.replace(/[^\n]/g, "")` keeps newlines intact and zeroes out the rest).

**Schema-introspection redact-coverage test (NEW — append to tests/unit/logger.test.ts):**

```typescript
// Plan 02.1-36 — closes UAT-NOTES.md §5.10. Derive expected redact paths
// from schema introspection so any future ciphertext column added without
// a corresponding REDACT_PATHS entry fails this test loudly (privacy floor
// per AGENTS.md item 6).
//
// Approach: scan src/lib/server/db/schema/*.ts for `bytea("<snake>")`
// declarations, extract the camelCase Drizzle field name from the same line,
// build the expected set, assert REDACT_PATHS contains `*.<camel>` and
// `*.<snake>` for every match.

it("Plan 02.1-36: REDACT_PATHS covers every ciphertext column in src/lib/server/db/schema/", async () => {
  const schemaDir = path.resolve(process.cwd(), "src", "lib", "server", "db", "schema");
  const entries = await fs.readdir(schemaDir, { withFileTypes: true });
  const expected = new Set<string>();
  for (const ent of entries) {
    if (!ent.isFile() || !ent.name.endsWith(".ts")) continue;
    const src = await fs.readFile(path.join(schemaDir, ent.name), "utf8");
    // Match: `<camel>: bytea("<snake>")` (Drizzle column declaration shape).
    // Tolerate whitespace and chained method calls (.notNull(), etc.).
    const re = /(\w+)\s*:\s*bytea\(\s*"([^"]+)"\s*\)/g;
    let match: RegExpExecArray | null;
    while ((match = re.exec(src)) !== null) {
      const camel = match[1]!;
      const snake = match[2]!;
      expected.add(`*.${camel}`);
      expected.add(`*.${snake}`);
    }
  }
  // Also include the already-covered KEK-related shapes that aren't bytea
  // columns (kek_version is smallint, not bytea; kek itself is module-level
  // constant). Discovered via grep for `kekVersion: smallint(...)` patterns.
  const kekVersionRe = /(\w+)\s*:\s*smallint\(\s*"(kek_version)"\s*\)/g;
  for (const ent of entries) {
    if (!ent.isFile() || !ent.name.endsWith(".ts")) continue;
    const src = await fs.readFile(path.join(schemaDir, ent.name), "utf8");
    let match: RegExpExecArray | null;
    while ((match = kekVersionRe.exec(src)) !== null) {
      expected.add(`*.${match[1]}`);
      expected.add(`*.${match[2]}`);
    }
  }
  // Cross-check against REDACT_PATHS exported from logger.ts. The export is
  // currently a const (not exported). Expose it via a test-only re-export OR
  // import the source file and parse it. Cleanest: ADD `export const
  // REDACT_PATHS = [...]` (currently `const REDACT_PATHS`) so the test can
  // import it directly. This is a non-load-bearing visibility tweak.
  const { REDACT_PATHS } = await import("$lib/server/logger.js");
  const actual = new Set(REDACT_PATHS);
  const missing = [...expected].filter((p) => !actual.has(p));
  expect(missing, `Missing REDACT_PATHS for ciphertext columns:\n${missing.join("\n")}`).toEqual([]);
});
```
